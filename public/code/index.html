        return m.reply(`Terjadi kesalahan, data tidak ditemukan.`)
    } 

        const title = TikTokDl.data.title;
        const videoUrl = TikTokDl.data.no_watermark;
        const audioUrl = TikTokDl.data.music;
        const videoCover = TikTokDl.data.cover;

    await bot.sendMessage(m.chat, {
        video: { url: videoUrl },
        caption: `*[ TikTok Video ]*\n\n${title}`,
        thumbnail: { url: videoCover },
    });
        
    await bot.sendMessage(m.chat, {
        audio: { url: audioUrl },
        caption: "Audio TikTok",
    });
    } catch (error) {
        console.log(error)
        m.reply(`Terjadi kesalahan: ${error.message || error}`)
    }
}
